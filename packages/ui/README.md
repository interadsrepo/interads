# `@interads/ui`

# Installation

```bash

  $ npm install @interads/ui styled-components tinycolor2

```

## Usage

```javascript

import { Button } from '@interads/ui'

export const Home: React.FC = function Home(props: HomeProps) {
  return (
    <section>
      <Button>This is Button</Button>
    </section>
  )
}

```
